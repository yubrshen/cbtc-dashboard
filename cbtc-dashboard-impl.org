        <label>messageType(s)</label>
        <choice value="*">ALL</choice>
        <default>*</default>
        <prefix>(</prefix>
        <suffix>)</suffix>
        <initialValue>*</initialValue>
        <delimiter> </delimiter>
        <search>
          <query>index="all_kafka"

          <earliest>$time.earliest$</earliest>
          <latest>$time.latest$</latest>
        </search>
        <fieldForLabel>messageType</fieldForLabel>
        <fieldForValue>messageType</fieldForValue>
        <valuePrefix>"</valuePrefix>
        <valueSuffix>"</valueSuffix>
      </input>
      <input type="multiselect" token="originators" searchWhenChanged="true">
        <label>originator(s)</label>
        <choice value="*">ALL</choice>
        <default>*</default>
        <prefix>(</prefix>
        <suffix>)</suffix>
        <initialValue>*</initialValue>
        <delimiter> </delimiter>
        <search>
          <query>index="all_kafka"
| spath input=_raw path=metadata.originator output=originator
| dedup originator | table originator</query>
          <earliest>$time.earliest$</earliest>
          <latest>$time.latest$</latest>
        </search>
        <fieldForLabel>originator</fieldForLabel>
        <fieldForValue>originator</fieldForValue>
        <valuePrefix>"</valuePrefix>
        <valueSuffix>"</valueSuffix>
      </input>
      <table>
        <search>
          <query>index="cbtc_messages" NOT icsbplantevent
| spath input=_raw path=metadata.messageTime output=messageTime
| spath input=_raw path=metadata.messageType output=messageType
| spath input=_raw path=metadata.originator output=originator

| eval epoch_messageTime= strptime(messageTime, "%Y-%m-%dT%H:%M:%S.%6NZ")

| search (messageType IN $messageTypes$) AND (originator IN $originators$)
| fields _time messageTime messageType originator</query>
          <earliest>$time.earliest$</earliest>
          <latest>$time.latest$</latest>
          <sampleRatio>1</sampleRatio>
          <refresh>10m</refresh>
          <refreshType>delay</refreshType>
        </search>
        <option name="drilldown">none</option>
        <option name="refresh.display">progressbar</option>
      </table>
